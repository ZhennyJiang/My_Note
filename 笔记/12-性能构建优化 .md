## 二、请求优化

### 1. 请求缓存 + 共享响应（解决数据丢失问题）

这是**最重要的优化**，解决你提到的"主应用和子应用同时请求同一接口时数据丢失"问题。

**文件**: `ys-common/packages/utils/src/requestCache.ts`（新建）

```typescript
interface CacheItem {
  data: any;
  timestamp: number;
  expiresIn: number;
}

// 响应缓存（各应用独立）
const responseCache = new Map<string, CacheItem>();

// 正在进行的请求（全局共享）- 挂载到 window 上解决主应用和子应用共享问题
const PENDING_PROMISES_KEY = '__QIANKUN_PENDING_PROMISES__';

const getPendingPromises = (): Map<string, Promise<any>> => {
  if (!(window as any)[PENDING_PROMISES_KEY]) {
    (window as any)[PENDING_PROMISES_KEY] = new Map();
  }
  return (window as any)[PENDING_PROMISES_KEY];
};

// 默认缓存时间 5 分钟
export const DEFAULT_CACHE_TIME = 5 * 60 * 1000;

/**
 * 生成缓存 key
 */
export const generateCacheKey = (
  method: string,
  url: string,
  params?: any,
  data?: any,
  appName?: string
): string => {
  let paramsStr = '';
  if (params) {
    paramsStr = JSON.stringify(params);
  } else if (data) {
    paramsStr = typeof data === 'string' ? data : JSON.stringify(data);
  }
  const base = `${method}-${url}-${paramsStr}`;
  return appName ? `${appName}-${base}` : base;
};

/**
 * 检查缓存是否有效
 */
const isCacheValid = (cached: CacheItem): boolean => {
  return Date.now() - cached.timestamp < cached.expiresIn;
};

/**
 * 获取缓存
 */
export const getCache = (key: string): any | null => {
  const cached = responseCache.get(key);
  if (cached && isCacheValid(cached)) {
    return cached.data;
  }
  if (cached) {
    responseCache.delete(key);
  }
  return null;
};

/**
 * 设置缓存
 */
export const setCache = (
  key: string,
  data: any,
  expiresIn: number = DEFAULT_CACHE_TIME
): void => {
  responseCache.set(key, {
    data,
    timestamp: Date.now(),
    expiresIn,
  });
};

/**
 * 清除所有缓存
 */
export const clearAllCache = (): void => {
  responseCache.clear();
};

/**
 * 清除指定 URL 的缓存
 */
export const clearCacheByUrl = (url: string): void => {
  responseCache.forEach((_, key) => {
    if (key.includes(url)) {
      responseCache.delete(key);
    }
  });
};

/**
 * 获取正在进行的请求 Promise
 */
export const getPendingPromise = (key: string): Promise<any> | undefined => {
  return getPendingPromises().get(key);
};

/**
 * 设置正在进行的请求 Promise
 */
export const setPendingPromise = (key: string, promise: Promise<any>): void => {
  getPendingPromises().set(key, promise);
};

/**
 * 删除正在进行的请求 Promise
 */
export const deletePendingPromise = (key: string): void => {
  getPendingPromises().delete(key);
};

/**
 * 清除所有正在进行的请求
 */
export const clearAllPendingPromises = (): void => {
  getPendingPromises().clear();
};
```

```typescript
import {
  generateCacheKey,
  getCache,
  setCache,
  clearCacheByUrl,
  getPendingPromise,
  setPendingPromise,
  deletePendingPromise,
  DEFAULT_CACHE_TIME,
} from '@ys-common/utils';

// 应用标识 - 用于隔离缓存
const APP_NAME = 'main-app';

// ...existing code... (保留原有的 request 实例和拦截器)

/**
 * @get请求 - 支持缓存 + 共享响应
 */
export function get(url: string, params: any = {}, config?: any): Promise<CustomAxiosResponse> {
  const requestConfig = {
    url,
    params,
    method: 'get',
    ...config,
  };

  const useCache = config?.cache !== false;
  const cacheTime = config?.cacheTime || DEFAULT_CACHE_TIME;

  // 带应用标识的缓存 key（用于隔离缓存）
  const cacheKey = generateCacheKey('get', url, params, null, APP_NAME);
  // 不带应用标识的 key（用于共享请求）
  const sharedKey = generateCacheKey('get', url, params);

  // 1. 优先检查本应用缓存
  if (useCache) {
    const cachedData = getCache(cacheKey);
    if (cachedData) {
      console.log(`[${APP_NAME}] 命中缓存: ${url}`);
      return Promise.resolve({
        data: cachedData,
        status: 200,
        statusText: 'OK (from cache)',
        headers: {},
        config: requestConfig,
      } as CustomAxiosResponse);
    }
  }

  // 2. 检查是否有相同的请求正在进行（跨应用共享响应）
  const pendingPromise = getPendingPromise(sharedKey);
  if (pendingPromise) {
    console.log(`[${APP_NAME}] 共享请求: ${url}`);
    return pendingPromise.then((response: any) => {
      // 共享响应后，也缓存到本应用
      if (useCache && (response.data?.code === 200 || response.data?.code === 0)) {
        setCache(cacheKey, response.data, cacheTime);
      }
      return response;
    });
  }

  // 3. 发起新请求
  const requestPromise = request(requestConfig)
    .then((response: any) => {
      // 缓存成功的响应
      if (useCache && (response.data?.code === 200 || response.data?.code === 0)) {
        setCache(cacheKey, response.data, cacheTime);
      }
      return response;
    })
    .finally(() => {
      // 请求完成后，从 pending 中移除
      deletePendingPromise(sharedKey);
    });

  // 将请求存入 pending，后续相同请求共享这个 Promise
  setPendingPromise(sharedKey, requestPromise);

  return requestPromise;
}

/**
 * @post请求 - 成功后清除相关缓存
 */
export function post(url: string, data?: any, config?: any): Promise<CustomAxiosResponse> {
  return request({
    url,
    data,
    method: 'post',
    ...config,
  }).then((res) => {
    // POST 成功后，清除相关 URL 的 GET 缓存
    if (config?.clearCache !== false && (res.data?.code === 200 || res.data?.code === 0)) {
      clearCacheByUrl(url.replace(/\/\d+$/, ''));
    }
    return res;
  });
}

/**
 * @put请求 - 成功后清除相关缓存
 */
export function put(url: any, data: any = {}, config?: any): Promise<CustomAxiosResponse> {
  return request({
    url,
    data,
    method: 'put',
    ...config,
  }).then((res) => {
    if (config?.clearCache !== false && (res.data?.code === 200 || res.data?.code === 0)) {
      clearCacheByUrl(url.replace(/\/\d+$/, ''));
    }
    return res;
  });
}

/**
 * @deletes请求 - 成功后清除相关缓存
 */
export function deletes(url: string, params: any = {}, config?: any): Promise<CustomAxiosResponse> {
  return request({
    url,
    params,
    method: 'delete',
    ...config,
  }).then((res) => {
    if (config?.clearCache !== false && (res.data?.code === 200 || res.data?.code === 0)) {
      clearCacheByUrl(url.replace(/\/\d+$/, ''));
    }
    return res;
  });
}

/**
 * @patch请求 - 成功后清除相关缓存
 */
export function patch(url: string, params: any = {}, config?: any): Promise<CustomAxiosResponse> {
  return request({
    url,
    data: params,
    method: 'patch',
    ...config,
  }).then((res) => {
    if (config?.clearCache !== false && (res.data?.code === 200 || res.data?.code === 0)) {

    return res;
  });
}
```

## 四、图片优化（华为云 OBS）

### 1. 图片处理工具函数

**文件**: `web-merchant-backend-base/src/utils/imageProcess.ts`（新建）

```typescript
interface ImageProcessOptions {
  format?: 'webp' | 'jpg' | 'png' | 'gif';
  quality?: number; // 1-100
  width?: number;
  height?: number;
  mode?: 'lfit' | 'mfit' | 'fill' | 'pad' | 'fixed';
}

/**
 * 判断是否是华为云 OBS 图片
 */
const isHuaweiObs = (url: string): boolean => {
  if (!url) return false;
  return (
    url.includes('myhuaweicloud.com') ||
    url.includes('obs.cn-') ||
    url.includes('broker-ys-common-static')
  );
};

/**
 * 华为云 OBS 图片处理
 */
export function processImage(url: string, options: ImageProcessOptions = {}): string {
  if (!url || !isHuaweiObs(url)) return url;

  const params: string[] = [];

  // 格式转换（默认 WebP）
  const format = options.format || 'webp';
  params.push(`format,${format}`);

  // 质量压缩
  if (options.quality && options.quality < 100) {
    params.push(`quality,q_${options.quality}`);
  }

  // 缩放
  if (options.width || options.height) {
    const mode = options.mode || 'lfit';
    let resizeParam = `resize,m_${mode}`;
    if (options.width) resizeParam += `,w_${options.width}`;
    if (options.height) resizeParam += `,h_${options.height}`;
    params.push(resizeParam);
  }

  if (params.length === 0) return url;

  const separator = url.includes('?') ? '&' : '?';
  return `${url}${separator}x-image-process=image/${params.join('/')}`;
}

/**
 * 获取缩略图
 */
export function getThumbnail(url: string, size: number = 200): string {
  return processImage(url, {
    format: 'webp',
    quality: 80,
    width: size,
    height: size,
    mode: 'fill',
  });
}

/**
 * 获取列表图
 */
export function getListImage(url: string, width: number = 400): string {
  return processImage(url, {
    format: 'webp',
    quality: 85,
    width,
  });
}

/**
 * 获取详情大图
 */
export function getDetailImage(url: string): string {
  return processImage(url, {
    format: 'webp',
    quality: 90,
  });
```

## 五、Nginx 服务端配置

**文件**: `nginx.conf`（部署时使用）

```
# 开启 Gzip 压缩
gzip on;
gzip_vary on;
gzip_min_length 1024;
gzip_proxied any;
gzip_types text/plain text/css text/xml text/javascript application/javascript application/x-javascript application/xml application/json;
gzip_comp_level 6;

# 开启 Brotli 压缩（比 Gzip 更高效，需要安装模块）
# brotli on;
# brotli_comp_level 6;
# brotli_types text/plain text/css text/xml text/javascript application/javascript application/json;

# 静态资源长期缓存
location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$ {
    expires 1y;
    add_header Cache-Control "public, immutable";
    add_header Vary Accept-Encoding;
}

# HTML 文件不缓存（保证获取最新版本）
location ~* \.html$ {
    expires -1;
    add_header Cache-Control "no-cache, no-store, must-revalidate";
}

# 子应用资源跨域配置
location /subapps/ {
    add_header Access-Control-Allow-Origin *;
    add_header Access-Control-Allow-Methods "GET, POST, OPTIONS";
    add_header Access-Control-Allow-Headers "DNT,X-CustomHeader,Keep-Alive,User-Agent,X-Requested-With,If-Modified-Since,Cache-Control,Content-Type";
}

# HTTP/2 配置（需要 HTTPS）
listen 443 ssl http2;
```



**为什么优化**：

- **Gzip/Brotli**：服务端压缩，减少传输体积 **60-70%**
- **静态资源缓存**：JS/CSS/图片缓存 1 年，二次访问直接使用缓存
- **HTML 不缓存**：保证用户获取最新版本
- **HTTP/2**：多路复用，减少请求等待时间
